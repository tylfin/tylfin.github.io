import math

highest_b = int(math.floor(1000/3))
for b in range(1, highest_b):
    a = (500000 - 1000*b)/(1000 - b)
    c = 1000 - b - a
    if c**2 == b**2 + a**2:
        print((a,b,c), a*b*c)

var highestB = Math.floor(1000/3.0);
for (var b = 1; b<highestB; b++) {
  var a = Math.round((500000.0 - 1000.0*b)/(1000.0 - b));
  var c = 1000.0 - a - b;
  if (a*a + b*b === c*c) {
    console.log([a,b,c,a*b*c]);
  }
}
